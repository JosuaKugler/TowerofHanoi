\begin{frame}{Gesetzmäßigkeit}
    \begin{definition}
    $M(n,k) := $ minimale Zugzahl bei $n$ Scheiben und $k$ Feldern
    \end{definition}
    \begin{table}[h]
        \centering
        \begin{tabular}{c|c}
            Scheiben & Benötigte Züge \\
            1 & 1 \\
            2 & 3 \\
            3 & 7 \\
        \end{tabular}
        \caption{Zugzahl bei 3 Feldern}
        \label{tab:3f}
    \end{table}
    \begin{alertblock}{Vermutung}
    \[M(n,3) = 2^n-1\]
    \end{alertblock}
\end{frame}
