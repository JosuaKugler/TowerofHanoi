\begin{frame}{Kryptographie}
    \begin{itemize}
        \item Wir verschlüsseln den Text \glqq Langer Text \grqq
        \item Dazu wählen wir 4 Parameter: $n$, $k$, \textit{step} und \textit{movecount}
    \end{itemize}
    \begin{myexample}
        $n=7$, $k=4$, \textit{step} = 4, \textit{movecount} = 15
    \end{myexample}
\end{frame}
